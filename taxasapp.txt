<script>
    const fields = ['vendaBruta', 'taxaProc', 'taxaGate', 'taxaGar', 'taxaAnte', 'taxaSaque'];
    
    function format(v) {
        return v.toLocaleString('pt-BR', { style: 'currency', currency: 'BRL' });
    }

    function update() {
        const val = {};
        fields.forEach(f => val[f] = parseFloat(document.getElementById(f).value) || 0);

        // Atualiza Labels das Regras
        document.getElementById('lProc').innerText = val.taxaProc;
        document.getElementById('lGar').innerText = val.taxaGar;
        document.getElementById('lAnte').innerText = val.taxaAnte;
        document.getElementById('lSaque').innerText = val.taxaSaque.toFixed(2);

        // Cálculos
        const dProc = val.vendaBruta * (val.taxaProc / 100);
        const liqAppmax = val.vendaBruta - dProc - val.taxaGate;
        const dGar = val.vendaBruta * (val.taxaGar / 100);
        const saldoSobra = liqAppmax - dGar;
        const dAnte = saldoSobra * (val.taxaAnte / 100);

        const res30 = saldoSobra - val.taxaSaque;
        const resAnt = saldoSobra - dAnte - val.taxaSaque;

        // Preenche Tabela
        document.getElementById('vBruto30').innerText = format(val.vendaBruta);
        document.getElementById('vBrutoAnt').innerText = format(val.vendaBruta);
        
        document.getElementById('vProc30').innerText = '- ' + format(dProc);
        document.getElementById('vProcAnt').innerText = '- ' + format(dProc);
        
        document.getElementById('vGate30').innerText = '- ' + format(val.taxaGate);
        document.getElementById('vGateAnt').innerText = '- ' + format(val.taxaGate);
        
        document.getElementById('vLiq30').innerText = format(liqAppmax);
        document.getElementById('vLiqAnt').innerText = format(liqAppmax);
        
        document.getElementById('vGar30').innerText = '- ' + format(dGar);
        document.getElementById('vGarAnt').innerText = '- ' + format(dGar);
        
        document.getElementById('vAnteAnt').innerText = '- ' + format(dAnte);
        
        document.getElementById('vSaque30').innerText = '- ' + format(val.taxaSaque);
        document.getElementById('vSaqueAnt').innerText = '- ' + format(val.taxaSaque);
        
        document.getElementById('vFinal30').innerText = format(res30);
        document.getElementById('vFinalAnt').innerText = format(resAnt);
        
        document.getElementById('vDiferenca').innerText = format(res30 - resAnt);
    }

    fields.forEach(f => document.getElementById(f).addEventListener('input', update));
    update();

    function gerarPDF() {
        const element = document.getElementById('pdf-report');
        const opt = {
            margin: [0, 0, 0, 0],
            filename: 'Relatorio_Financeiro_Appmax.pdf',
            image: { type: 'jpeg', quality: 1 },
            html2canvas: { scale: 3, useCORS: true },
            jsPDF: { unit: 'mm', format: 'a4', orientation: 'portrait' }
        };
        html2pdf().set(opt).from(element).save();
    }
</script>
